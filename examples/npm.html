NPM Usage Examples

1. Installation

npm install gsap-dattebayo gsap

2. Import and Use

// Import specific animations
import { fadeUp, zoomIn, charReveal } from 'gsap-dattebayo';

// Animate elements
fadeUp('.title', { duration: 1.2 });
zoomIn('.card', { stagger: 0.1 });
charReveal('.heading', { stagger: 0.05 });

3. Auto-Init with Data Attributes

import { init } from 'gsap-dattebayo';

// Initialize with auto-detection
init({
  autoDetect: true,
  defaults: {
    duration: 1,
    ease: 'power2.out'
  }
});

Then use data attributes in HTML:

<div data-gsap="fadeUp">Content</div>
<h1 data-gsap="charReveal" data-gsap-stagger="0.05">Title</h1>

4. Scroll Animations

import { parallax, scrollReveal } from 'gsap-dattebayo';

// Parallax effect
parallax('.background', {
  speed: 0.5,
  direction: 'vertical'
});

// Scroll reveal
scrollReveal('.card', {
  animation: 'fadeUp',
  start: 'top 80%',
  once: true
});

5. Text Animations (SplitText)

import {
  charReveal,
  wordReveal,
  lineReveal,
  scrambleReveal
} from 'gsap-dattebayo';

// Character-by-character
charReveal('.title', {
  stagger: 0.05,
  ease: 'back.out(1.7)'
});

// Word-by-word
wordReveal('.subtitle', {
  stagger: 0.1
});

// Scramble effect
scrambleReveal('.cyberpunk-text', {
  duration: 2,
  chars: '01'
});

6. React Usage

import { useEffect } from 'react';
import { fadeUp, init } from 'gsap-dattebayo';

function App() {
  useEffect(() => {
    // Initialize on mount
    init({ autoDetect: true });

    // Or manually animate
    fadeUp('.element');

    // Cleanup
    return () => {
      // GSAP auto-cleanup
    };
  }, []);

  return (
    <div data-gsap="fadeUp">
      Animated Content
    </div>
  );
}
